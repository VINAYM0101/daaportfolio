// Function for Depth-First Search (DFS)
void dfs(int node, vector>& adj, vector& visited) {
    visited[node] = true;
    cout << node << " ";
    for (int neighbor : adj[node]) {
        if (!visited[neighbor]) {
            dfs(neighbor, adj, visited);
        }
    }
}

// Function for Breadth-First Search (BFS)
void bfs(int start, vector>& adj, int n) {
    vector visited(n, false);
    queue q;
    visited[start] = true;
    q.push(start);
    while (!q.empty()) {
        int node = q.front();
        q.pop();
        cout << node << " ";
        for (int neighbor : adj[node]) {
            if (!visited[neighbor]) {
                visited[neighbor] = true;
                q.push(neighbor);
            }
        }
    }
}

// Function for Dijkstra's Algorithm
void dijkstra(int source, vector>& adjMatrix, int n) {
    vector dist(n, INT_MAX);
    dist[source] = 0;
    priority_queue, vector>, greater>> pq;
    pq.push({0, source});

    while (!pq.empty()) {
        int u = pq.top().second;
        pq.pop();

        for (int v = 0; v < n; ++v) {
            if (adjMatrix[u][v] != INT_MAX && dist[u] + adjMatrix[u][v] < dist[v]) {
                dist[v] = dist[u] + adjMatrix[u][v];
                pq.push({dist[v], v});
            }
        }
    }

    for (int i = 0; i < n; ++i) {
        cout << "Node " << i << ": " << dist[i] << endl;
    }
}

// Function for Prim's Algorithm (Minimum Spanning Tree)
void primsMST(vector>& adjMatrix, int n) {
    vector inMST(n, false);
    vector key(n, INT_MAX);
    vector parent(n, -1);
    key[0] = 0;

    for (int count = 0; count < n - 1; ++count) {
        int u = -1;
        for (int i = 0; i < n; ++i) {
            if (!inMST[i] && (u == -1 || key[i] < key[u])) {
                u = i;
            }
        }
        inMST[u] = true;

        for (int v = 0; v < n; ++v) {
            if (adjMatrix[u][v] != INT_MAX && !inMST[v] && adjMatrix[u][v] < key[v]) {
                key[v] = adjMatrix[u][v];
                parent[v] = u;
            }
        }
    }

    cout << "MST Edges:" << endl;
    for (int i = 1; i < n; ++i) {
        cout << parent[i] << " - " << i << endl;
    }
}

// Function for QuickSort
void quicksort(vector& arr, int low, int high) {
    if (low < high) {
        int pivot = arr[high];
        int i = low - 1;
        for (int j = low; j < high; ++j) {
            if (arr[j] <= pivot) {
                ++i;
                swap(arr[i], arr[j]);
            }
        }
        swap(arr[i + 1], arr[high]);
        int pi = i + 1;
        quicksort(arr, low, pi - 1);
        quicksort(arr, pi + 1, high);
    }
}

// Bellman-Ford Algorithm (for negative weight cycle detection)
void bellmanFord(vector>& graph, int V, int E, int src) {
    vector dist(V, INT_MAX);
    dist[src] = 0;

    for (int i = 1; i < V; ++i) {
        for (int j = 0; j < E; ++j) {
            if (dist[graph[j][0]] != INT_MAX && dist[graph[j][0]] + graph[j][2] < dist[graph[j][1]]) {
                dist[graph[j][1]] = dist[graph[j][0]] + graph[j][2];
            }
        }
    }

    for (int i = 0; i < E; ++i) {
        if (dist[graph[i][0]] != INT_MAX && dist[graph[i][0]] + graph[i][2] < dist[graph[i][1]]) {
            cout << "Graph contains negative weight cycle" << endl;
            return;
        }
    }

    for (int i = 0; i < V; ++i) {
        cout << "Node " << i << ": " << dist[i] << endl;
    }
}

// Floyd-Warshall Algorithm (for all-pairs shortest path)
void floydWarshall(vector>& graph, int V) {
    vector> dist = graph;
    for (int k = 0; k < V; ++k) {
        for (int i = 0; i < V; ++i) {
            for (int j = 0; j < V; ++j) {
                if (dist[i][k] != INT_MAX && dist[k][j] != INT_MAX &&
                    dist[i][k] + dist[k][j] < dist[i][j]) {
                    dist[i][j] = dist[i][k] + dist[k][j];
                }
            }
        }
    }

    cout << "Shortest distances between every pair of nodes:" << endl;
    for (int i = 0; i < V; ++i) {
        for (int j = 0; j < V; ++j) {
            if (dist[i][j] == INT_MAX) cout << "INF ";
            else cout << dist[i][j] << " ";
        }
        cout << endl;
    }
}

// Kruskal's Algorithm (Minimum Spanning Tree)
struct Edge {
    int u, v, weight;
    bool operator<(const Edge& e) {
        return weight < e.weight;
    }
};

int findParent(int u, vector& parent) {
    if (parent[u] == u) return u;
    return parent[u] = findParent(parent[u], parent);
}

void kruskal(vector& edges, int V) {
    sort(edges.begin(), edges.end());
    vector parent(V);
    for (int i = 0; i < V; ++i) parent[i] = i;

    vector mst;

    for (Edge& edge : edges) {
        int u = findParent(edge.u, parent);
        int v = findParent(edge.v, parent);

        if (u != v) {
            mst.push_back(edge);
            parent[u] = v;
        }
    }

    cout << "Kruskal's MST Edges:" << endl;
    for (Edge& e : mst) {
        cout << e.u << " - " << e.v << " (" << e.weight << ")" << endl;
    }
}
